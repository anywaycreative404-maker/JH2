(function loop() {

  ring.style.left=rx+'px'; ring.style.top=ry+'px';

})();

const nameEl = document.getElementById('heroName');

lines.forEach((line,li)=>{

  const row = document.createElement('div');

  row.style.cssText = 'display:block;overflow:hidden;line-height:0.88';

  [...line.t].forEach((ch,ci)=>{

    const s = document.createElement('span');

    s.classList.add('ch');

    if(line.out) s.classList.add('out');

    s.textContent = ch;

    s.style.animationDelay = (1.6 + li*0.2 + ci*0.048) + 's';

    row.appendChild(s);

  });

  nameEl.appendChild(row);

});


/* REVEAL */

const obs = new IntersectionObserver(entries=>{

  entries.forEach((e,i)=>{

    if(e.isIntersecting) setTimeout(()=>e.target.classList.add('on'), i*55);

  });

},{threshold:0.1,rootMargin:'0px 0px -30px 0px'});

document.querySelectorAll('.reveal').forEach(el=>obs.observe(el));


/* MAGNETIC BUTTONS */

document.querySelectorAll('[data-mag]').forEach(btn=>{

  btn.addEventListener('mousemove',e=>{

    const r=btn.getBoundingClientRect();

    const dx=(e.clientX - r.left - r.width/2)*0.3;

    const dy=(e.clientY - r.top  - r.height/2)*0.3;

    btn.style.transition='transform .1s ease';

    btn.style.transform=`translate(${dx}px,${dy}px)`;

  });

  btn.addEventListener('mouseleave',()=>{

    btn.style.transition='transform .55s cubic-bezier(.16,1,.3,1)';

    btn.style.transform='translate(0,0)';

  });

});
